DECLARE
    v_roll NUMBER;
    v_attendance NUMBER(5,2);
    v_status VARCHAR2(20);

BEGIN
    v_roll := &roll_no;

    SELECT Attendance INTO v_attendance
    FROM Stud
    WHERE Roll = v_roll;

    IF v_attendance < 75 THEN
        v_status := 'Detained';
        DBMS_OUTPUT.PUT_LINE('Term not granted');
    ELSE
        v_status := 'Not Detained';
        DBMS_OUTPUT.PUT_LINE('Term granted');
    END IF;

    UPDATE Stud
    SET Status = v_status
    WHERE Roll = v_roll;

    COMMIT;

END;
